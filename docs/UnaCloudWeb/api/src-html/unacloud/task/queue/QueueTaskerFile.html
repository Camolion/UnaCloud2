001package unacloud.task.queue;
002
003import unacloud.share.enums.QueueMessageType;
004import unacloud.share.queue.QueueMessage;
005import unacloud.share.queue.QueueTaskerConnection;
006import unacloud.User;
007import unacloud.VirtualMachineImage;
008
009/**
010 * Class used to put task in queue messaging service that will be read by File Manager project 
011 * @author CesarF
012 *
013 */
014public class QueueTaskerFile {
015        
016        /**
017         * Represents class to connect to queue provider
018         */
019        private static QueueTaskerConnection fileQueue;
020        
021        /**
022         * Sets the queue manager used to send task. This method should only be called one time.
023         * @param newQueue
024         */
025        public static void setQueueConnection(QueueTaskerConnection newQueue){
026                fileQueue = newQueue;
027        }
028        
029        /**
030         * Creates a task to copy an image from a private folder to public one.
031         * @param image to copy
032         * @param user image owner
033         */
034        public static void createPublicCopy(VirtualMachineImage image, User user){
035                QueueMessage message = new QueueMessage(QueueMessageType.CREATE_PUBLIC_IMAGE, user.getDatabaseId()+"", new String[]{image.getDatabaseId()+""});
036                fileQueue.sendMessage(message);
037        }
038        /**
039         * Creates a task to copy a public image to a private one.
040         * @param publicImage
041         * @param image
042         * @param user
043         */
044        public static void createCopyFromPublic(VirtualMachineImage publicImage, VirtualMachineImage image, User user){
045                QueueMessage message = new QueueMessage(QueueMessageType.CREATE_COPY_FROM_PUBLIC, user.getDatabaseId()+"", new String[]{image.getDatabaseId()+"",publicImage.getDatabaseId()+""});
046                fileQueue.sendMessage(message);
047        }
048        /**
049         * Creates a task to delete image from repositories
050         * @param image
051         * @param user
052         */
053        public static void deleteImage(VirtualMachineImage image, User user){
054                QueueMessage message = new QueueMessage(QueueMessageType.DELETE_IMAGE, user.getDatabaseId()+"", new String[]{image.getDatabaseId()+""});
055                fileQueue.sendMessage(message);
056        }
057        
058        /**
059         * Creates a task to delete public image from repositories
060         * @param image
061         * @param user
062         */
063        public static void deletePublicImage(VirtualMachineImage image, User user){
064                QueueMessage message = new QueueMessage(QueueMessageType.DELETE_PUBLIC_IMAGE, user.getDatabaseId()+"", new String[]{image.getDatabaseId()+""});
065                fileQueue.sendMessage(message);
066        }
067        
068        /**
069         * Puts a task to remove an user, its machines, clusters and deployments
070         * @param user User that will be removed
071         */
072        public static void deleteUser(User user, User admin){
073                QueueMessage message = new QueueMessage(QueueMessageType.DELETE_USER, admin.getDatabaseId()+"", new String[]{user.getDatabaseId()+""});
074                fileQueue.sendMessage(message);
075        }
076        
077}
